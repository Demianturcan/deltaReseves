CREATE DATABASE reservadb;
CREATE USER 'reservadb'@'localhost' IDENTIFIED BY 'microdelta';
GRANT ALL PRIVILEGES ON reservadb.* TO 'reservadb'@'localhost';


CREATE TABLE reservas (
    id INT PRIMARY KEY,
    nombre_cliente INT,
    numero_personas INT NOT NULL,
    hora_reserva TIME NOT NULL,
    estado VARCHAR(50) NOT NULL
);
